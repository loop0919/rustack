#フィボナッチ数列の値を計算する
$fib {
    $n exch def
    { n 1 < }
    { 0 }
    {
        { n 2 < }
        { 1 }
        {
            n 1 -
            fib
            n 2 -
            fib
            +
        }
        if
    }
    if
} def

$call {
    $x exch def
    { x 1 < }
    { }
    { 
        x fib puts 
        x 1 + call
    } if
} def

15 call
